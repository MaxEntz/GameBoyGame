\chapter{Architecture and Module Documentation}

\section{Overall Architecture}

The program starts in \texttt{main}, calls \texttt{core}, initializes shared text and dialogue assets, registers an empty VBlank ISR, loads the initial menu, and then runs a frame loop. Each frame waits for VBlank, changes scene if requested, reads the joypad, dispatches input to the current scene, and dispatches the current scene update.

The scene table in \texttt{src/main/core.c} maps \texttt{game\_state\_t} values to three callbacks: load assets, handle input, and update. Implemented states are \texttt{GAME\_STATE\_MENU}, \texttt{GAME\_STATE\_SAVE\_SELECT}, \texttt{GAME\_STATE\_CHOOSE\_DIFFICULTY}, \texttt{GAME\_STATE\_LOBBY}, \texttt{GAME\_STATE\_MG1}, \texttt{GAME\_STATE\_MG2}, and \texttt{GAME\_STATE\_MG3}.

Cycle costs are not documented in the codebase.

\section{Main Module}

\subsection{Source Files}

\begin{description}
  \item[\texttt{src/main/main.c}] C entry point; calls \texttt{core}.
  \item[\texttt{src/main/core.c}] Global scene table, display toggles, VBlank registration, input dispatch, update dispatch, and main loop.
\end{description}

\subsection*{\texttt{main}}
\begin{description}
  \item[Input] None.
  \item[Output] None.
  \item[Side effects] Enters the game by calling \texttt{core}.
  \item[Calls] \texttt{core}.
\end{description}

\subsection*{\texttt{core}}
\begin{description}
  \item[Input] None.
  \item[Output] None.
  \item[Side effects] Initializes \texttt{game\_t}, loads shared VRAM assets, installs the VBlank ISR, enables interrupts, controls LCD visibility, and runs until \texttt{game.is\_running} becomes false.
  \item[Calls] \texttt{text\_renderer\_init}, \texttt{dialogue\_init}, \texttt{add\_VBL}, \texttt{enable\_interrupts}, \texttt{wait\_vbl\_done}, \texttt{change\_scene}, scene input callbacks, scene update callbacks.
\end{description}

\subsection*{\texttt{change\_scene}}
\begin{description}
  \item[Input] \texttt{game} points to the global runtime state. \texttt{game->state\_changed} determines whether work is done.
  \item[Output] Clears \texttt{game->state\_changed} after loading the new state.
  \item[Side effects] Turns display, sprites, background, and window off; calls the current state's asset loader; then turns display, sprites, and background on.
  \item[Calls] Current scene load callback.
\end{description}

\section{Common Modules}

\subsection{Source Files and State}

\begin{description}
  \item[\texttt{src/common/text\_renderer.c}] Shared text renderer using the common font at tile base 128. Owns \texttt{g\_text\_renderer\_ready}.
  \item[\texttt{src/common/dialogue.c}] Dialogue box state machine and typing sound. Owns \texttt{g\_dialogue\_ready}.
  \item[\texttt{src/common/game\_changer.c}] State transitions with fade, sound, optional VRAM clearing, and immediate scene load.
  \item[\texttt{src/common/save.c}] Battery-backed save load, write, and reset at \texttt{\$A000}.
  \item[\texttt{src/common/transition.c}] Palette fade and channel 1 transition sound effects.
  \item[\texttt{src/common/random.c}] RNG seeding and modulo random helper. Owns \texttt{g\_seed}.
  \item[\texttt{src/common/get\_tile\_by\_map.c}] Pixel-to-tile lookup helper for 20x18 maps.
  \item[\texttt{src/common/vram\_clear.c}] Clears 256 background tiles, background map, text/dialogue readiness flags, and all 40 sprites.
\end{description}

\subsection*{\texttt{text\_renderer\_init}}
\begin{description}
  \item[Input] None.
  \item[Output] None.
  \item[Side effects] Loads \texttt{font\_tiles} to VRAM starting at tile \texttt{128}; marks renderer ready.
  \item[Calls] \texttt{set\_bkg\_data}.
\end{description}

\subsection*{\texttt{text\_renderer\_destroy}}
\begin{description}
  \item[Input] None.
  \item[Output] None.
  \item[Side effects] Allows the font tiles to be reloaded after VRAM clearing.
  \item[Calls] None.
\end{description}

\subsection*{\texttt{text\_renderer\_draw}}
\begin{description}
  \item[Input] \texttt{render} points to \texttt{text\_render\_t}; \texttt{text} must be null-terminated; \texttt{x} and \texttt{y} are tile coordinates.
  \item[Output] None.
  \item[Side effects] Writes font tile IDs to the background map until the string ends or the row reaches tile X 20.
  \item[Calls] \texttt{set\_bkg\_tile\_xy}.
\end{description}

\subsection*{\texttt{text\_renderer\_clear}}
\begin{description}
  \item[Input] Same contract as \texttt{text\_renderer\_draw}.
  \item[Output] None.
  \item[Side effects] Replaces the rendered text range with the space tile.
  \item[Calls] \texttt{set\_bkg\_tile\_xy}.
\end{description}

\subsection*{\texttt{dialogue\_init}}
\begin{description}
  \item[Input] None.
  \item[Output] None.
  \item[Side effects] Loads dialogue box tiles into VRAM at tile \texttt{171}; marks dialogue assets ready.
  \item[Calls] \texttt{set\_bkg\_data}.
\end{description}

\subsection*{\texttt{dialogue\_destroy}}
\begin{description}
  \item[Input] None.
  \item[Output] None.
  \item[Side effects] Allows dialogue tiles to be reloaded after VRAM clearing.
  \item[Calls] None.
\end{description}

\subsection*{\texttt{dialogue\_start}}
\begin{description}
  \item[Input] \texttt{dlg} points to a \texttt{dialogue\_t}; \texttt{text} is a null-terminated string where \texttt{\textbackslash n} separates lines.
  \item[Output] Initializes the dialogue state to page 0 and animation state.
  \item[Side effects] Writes the dialogue window tilemap, moves the window to Y 112, and shows the window.
  \item[Calls] \texttt{set\_win\_tiles}, \texttt{move\_win}, \texttt{SHOW\_WIN}.
\end{description}

\subsection*{\texttt{dialogue\_update}}
\begin{description}
  \item[Input] \texttt{dlg} points to active dialogue state.
  \item[Output] Advances typing timers and state.
  \item[Side effects] Writes one character tile at a time to the window map and plays channel 1 typing sound for non-space characters.
  \item[Calls] \texttt{set\_win\_tile\_xy}; internal sound helper writes \texttt{NR10\_REG} through \texttt{NR14\_REG}, \texttt{NR50\_REG}, \texttt{NR51\_REG}, and \texttt{NR52\_REG}.
\end{description}

\subsection*{\texttt{dialogue\_handle\_input}}
\begin{description}
  \item[Input] \texttt{dlg} points to dialogue state; \texttt{keys} is the current \texttt{joypad()} mask.
  \item[Output] May skip animation, advance a page, or close the dialogue.
  \item[Side effects] Updates \texttt{dlg}; may hide the window.
  \item[Calls] Internal page helpers and \texttt{HIDE\_WIN}.
\end{description}

\subsection*{\texttt{dialogue\_is\_active}}
\begin{description}
  \item[Input] \texttt{dlg} points to dialogue state.
  \item[Output] Returns true when the dialogue state is not inactive.
  \item[Side effects] None.
  \item[Calls] None.
\end{description}

\subsection*{\texttt{game\_changer}}
\begin{description}
  \item[Input] \texttt{game} points to runtime state; \texttt{new\_state} is the requested state; \texttt{clear\_vram} decides whether to clear VRAM.
  \item[Output] Updates \texttt{game->state} and \texttt{game->state\_changed} if valid.
  \item[Side effects] Plays transition sound, fades out and in, optionally clears VRAM, loads the new scene immediately via \texttt{change\_scene}, and delays 300 ms.
  \item[Calls] \texttt{transition\_sound}, \texttt{transition\_fade\_out}, \texttt{vram\_clear}, \texttt{change\_scene}, \texttt{delay}, \texttt{transition\_fade\_in}.
\end{description}

\subsection*{\texttt{save\_load}}
\begin{description}
  \item[Input] \texttt{game} points to runtime state. SRAM must contain \texttt{save\_data\_t} or an invalid magic value.
  \item[Output] Updates scores, lobby position, lobby map, dialogue index, and difficulty if \texttt{SAVE\_MAGIC} matches.
  \item[Side effects] Enables and disables external RAM; reads from \texttt{\$A000}.
  \item[Calls] \texttt{lobby\_get\_state}, \texttt{set\_map}, \texttt{memcpy}.
\end{description}

\subsection*{\texttt{save\_write}}
\begin{description}
  \item[Input] \texttt{game} points to runtime state; lobby state is available through \texttt{lobby\_get\_state}.
  \item[Output] Updates best scores in \texttt{game} if current scores are higher.
  \item[Side effects] Writes \texttt{save\_data\_t} to external RAM at \texttt{\$A000}.
  \item[Calls] \texttt{lobby\_get\_state}, \texttt{memcpy}.
\end{description}

\subsection*{\texttt{save\_reset}}
\begin{description}
  \item[Input] \texttt{game} provides current best scores.
  \item[Output] None.
  \item[Side effects] Writes a fresh save at \texttt{\$A000}, preserving best scores and resetting run scores, dialogue index, player position, map, and difficulty.
  \item[Calls] \texttt{memcpy}.
\end{description}

\subsection*{\texttt{transition\_fade\_out}}
\begin{description}
  \item[Input] None.
  \item[Output] None.
  \item[Side effects] Writes \texttt{BGP\_REG} through four palettes from normal to dark and plays transition tones.
  \item[Calls] \texttt{wait\_vbl\_done}, \texttt{transition\_sound}, \texttt{delay}.
\end{description}

\subsection*{\texttt{transition\_fade\_in}}
\begin{description}
  \item[Input] None.
  \item[Output] None.
  \item[Side effects] Writes \texttt{BGP\_REG} through four palettes from dark to normal and plays transition tones.
  \item[Calls] \texttt{wait\_vbl\_done}, \texttt{transition\_sound}, \texttt{delay}.
\end{description}

\subsection*{\texttt{transition\_sound}}
\begin{description}
  \item[Input] \texttt{stage} selects one of four \texttt{NR13} frequency low values; invalid stage resets to 0.
  \item[Output] None.
  \item[Side effects] Enables sound and triggers channel 1.
  \item[Calls] None.
\end{description}

\subsection*{\texttt{random\_init}}
\begin{description}
  \item[Input] \texttt{keys} is the current nonzero joypad mask.
  \item[Output] Seeds GBDK's RNG.
  \item[Side effects] Reads \texttt{DIV\_REG}; writes static \texttt{g\_seed}.
  \item[Calls] \texttt{initrand}.
\end{description}

\subsection*{\texttt{random\_get}}
\begin{description}
  \item[Input] \texttt{modulo} is used as the divisor for \texttt{rand() \% modulo}; callers must pass nonzero.
  \item[Output] Returns a \texttt{UINT8} in the range \texttt{0} to \texttt{modulo - 1}.
  \item[Side effects] Advances the GBDK RNG state.
  \item[Calls] \texttt{rand}.
\end{description}

\subsection*{\texttt{get\_tile\_by\_map}}
\begin{description}
  \item[Input] \texttt{map} is a 20x18 tile array; \texttt{x} and \texttt{y} are pixel coordinates.
  \item[Output] Returns the tile ID at \texttt{x / 8}, \texttt{y / 8}; returns 0 out of bounds.
  \item[Side effects] None.
  \item[Calls] None.
\end{description}

\subsection*{\texttt{vram\_clear}}
\begin{description}
  \item[Input] None.
  \item[Output] None.
  \item[Side effects] Clears 256 background tile patterns to zero, invalidates shared text/dialogue loaded flags, fills the visible background map with tile 0, and moves all 40 sprites to 0,0.
  \item[Calls] \texttt{set\_bkg\_data}, \texttt{text\_renderer\_destroy}, \texttt{dialogue\_destroy}, \texttt{fill\_bkg\_rect}, \texttt{move\_sprite}.
\end{description}

\section{Lobby Modules}

\subsection{Source Files and State}

\begin{description}
  \item[\texttt{src/lobby/lobby.c}] Lobby runtime state, asset loading, player sprite update, NPC animation, and frame counters. Owns static \texttt{g\_lobby}.
  \item[\texttt{src/lobby/input\_handler\_lobby.c}] Movement, collision, NPC interaction, RNG initialization, save-on-select, and menu return.

\subsection*{\texttt{lobby\_init\_state}}
\begin{description}
  \item[Input] None.
  \item[Output] Initializes \texttt{g\_lobby} to spawn position, map \texttt{MAP\_ID\_BL}, default movement direction, counters, and dialogue state.
  \item[Side effects] Copies map data through \texttt{set\_map}.
  \item[Calls] \texttt{set\_map}.
\end{description}

\subsection*{\texttt{lobby\_get\_state}}
\begin{description}
  \item[Input] None.
  \item[Output] Returns pointer to static \texttt{g\_lobby}.
  \item[Side effects] None.
  \item[Calls] None.
\end{description}

\subsection*{\texttt{lobby}}
\begin{description}
  \item[Input] \texttt{game} points to runtime state.
  \item[Output] May load save data into \texttt{game} and lobby state the first time.
  \item[Side effects] Loads lobby tiles, sprite data, current map, text/dialogue assets, and scoreboard if on \texttt{MAP\_ID\_TC}.
  \item[Calls] \texttt{lobby\_init\_state}, \texttt{save\_load}, \texttt{set\_bkg\_data}, \texttt{set\_bkg\_tiles}, \texttt{set\_sprite\_data}, \texttt{set\_sprite\_tile}, \texttt{move\_sprite}, \texttt{text\_renderer\_init}, \texttt{dialogue\_init}, \texttt{draw\_scoreboard}.

\subsection*{\texttt{handle\_input\_lobby}}
\begin{description}
  \item[Input] \texttt{game}; \texttt{keys} from \texttt{joypad()}.
  \item[Output] Updates lobby player position and movement direction.
  \item[Side effects] Seeds RNG on first key press, advances dialogue input, starts NPC dialogue with A, writes save and returns to menu on Select, and may switch maps.
  \item[Calls] \texttt{random\_init}, \texttt{dialogue\_is\_active}, \texttt{dialogue\_handle\_input}, \texttt{lore\_start\_dialogue}, \texttt{lore\_start\_idle\_dialogue}, \texttt{save\_write}, \texttt{game\_changer}, \texttt{get\_tile\_by\_map}, \texttt{switch\_map}.
\end{description}

\subsection*{\texttt{update\_lobby}}
\begin{description}
  \item[Input] \texttt{game} points to runtime state.
  \item[Output] Updates lobby counters and sprite position.
  \item[Side effects] Animates NPC tiles, updates lore/dialogue, updates four player sprites.
  \item[Calls] \texttt{lore\_update}, \texttt{set\_bkg\_data}, \texttt{set\_sprite\_tile}, \texttt{move\_sprite}.
\end{description}

\subsection*{\texttt{transition\_map\_animation}}
\begin{description}
  \item[Input] \texttt{game}; \texttt{dir} is \texttt{TRANSITION\_TB}, \texttt{TRANSITION\_BT}, \texttt{TRANSITION\_LR}, or \texttt{TRANSITION\_RL}.
  \item[Output] Updates lobby player coordinates after crossing the map boundary.
  \item[Side effects] Displays \texttt{transition\_map}, moves player sprites in a blocking loop, waits for VBlank per step, then restores the new current map and scoreboard if needed.
  \item[Calls] \texttt{set\_bkg\_tiles}, \texttt{move\_sprite}, \texttt{wait\_vbl\_done}, \texttt{draw\_scoreboard}.
\end{description}

\subsection*{\texttt{switch\_map}}
\begin{description}
  \item[Input] \texttt{game}; \texttt{x} and \texttt{y} are pixel collision coordinates; \texttt{tile} is the touched tile.
  \item[Output] Returns true if a map switch starts.
  \item[Side effects] Updates current map ID and map copy, marks map as changing, and runs transition animation.
  \item[Calls] \texttt{transition\_map\_animation}.

\subsection*{\texttt{set\_map}}

\subsection*{\texttt{lore\_start\_idle\_dialogue}}

  \item[Output] Updates current piece rotation, horizontal position, and soft-drop timer.

  \item[Calls] \texttt{game\_changer}, \texttt{rotate\_r}, \texttt{rotate\_l}, \texttt{piece\_can\_move\_left}, \texttt{piece\_can\_move\_right}, \texttt{piece\_erase}, \texttt{piece\_draw}.

\subsection*{\texttt{update\_tetris}}

  \item[Calls] \texttt{piece\_can\_move\_down}, \texttt{piece\_erase}, \texttt{piece\_draw}, \texttt{piece\_lock}, \texttt{grid\_clear\_lines}, \texttt{grid\_draw\_playfield}, \texttt{draw\_score}, \texttt{draw\_lines}, \texttt{draw\_level}, \texttt{draw\_next\_piece}, \texttt{piece\_can\_spawn}, \texttt{game\_changer}.
